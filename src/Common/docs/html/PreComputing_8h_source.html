193 /*
194  * @brief Compute pseudo-inverse for trivial matrix for B-Metric clustering
195  *
196  * @param[in] a The input matrix
197  * @param[in] epsilon The epsilon as the accuracy for the pseudo-inverse computation
198  */
199 template<typename _Matrix_Type_>
200 _Matrix_Type_ pseudoInverse(const _Matrix_Type_ &a,
201  double epsilon = std::numeric_limits<double>::epsilon())
202 {
203  Eigen::JacobiSVD< _Matrix_Type_ > svd(a ,Eigen::ComputeThinU | Eigen::ComputeThinV);
204  double tolerance = epsilon * std::max(a.cols(), a.rows())
205  *svd.singularValues().array().abs()(0);
206  return svd.matrixV() * (svd.singularValues().array().abs() > tolerance).select
207  (svd.singularValues().array().inverse(), 0).matrix().asDiagonal()
208  * svd.matrixU().adjoint();
209 }
