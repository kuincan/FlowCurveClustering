293 /*
294  * @brief The template for computing numerical integration
295  * @details
296  * The implementation by boost C++ library can be seed at
297  * https://www.boost.org/doc/libs/1_63_0/libs/multiprecision/doc/html/boost_multiprecision/tut/floats/fp_eg/gi.html
298  *
299  * @param[in] a The value type of a
300  * @param[in] b The value type of b
301  * @param[in] tol The step size
302  * @param[in] func The function type
303  */
304 template<typename value_type, typename function_type>
305 inline value_type integral(const value_type a,
306  const value_type b,
307  const value_type tol,
308  function_type func)
309 {
310  unsigned n = 1U;
311 
312  value_type h = (b - a);
313  value_type I = (func(a) + func(b)) * (h / 2);
314 
315  for(unsigned k = 0U; k < 8U; k++)
316  {
317  h /= 2;
318 
319  value_type sum(0);
320  for(unsigned j = 1U; j <= n; j++)
321  {
322  sum += func(a + (value_type((j * 2) - 1) * h));
323  }
324 
325  const value_type I0 = I;
326  I = (I / 2) + (h * sum);
327 
328  const value_type ratio = I0 / I;
329  const value_type delta = ratio - 1;
330  const value_type delta_abs = ((delta < 0) ? -delta : delta);
331 
332  if((k > 1U) && (delta_abs < tol))
333  {
334  break;
335  }
336 
337  n *= 2U;
338  }
339 
340  return I;
341 }
